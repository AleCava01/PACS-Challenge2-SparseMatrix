38// Updates the value at position (i, j) in the uncompressed (sparse_data_) format.

50template<typename T, StorageOrder Order>
51void Matrix<T, Order>::compress() {
52// Converts the matrix from uncompressed (Coo-MAP) format to compressed format (CSR for RowMajor, CSC for ColMajor).
53// It builds compressed_data_ from sparse_data_ and then clears the uncompressed storage to save memory.
54
55 // Determine the conversion type
56 constexpr bool isRowMajor = (Order == StorageOrder::RowMajor);
57 size_t outer_size = isRowMajor ? rows_ : cols_;
58
59 // Clears the values and preparation
60 compressed_data_.values.clear();
61 compressed_data_.inner_index.clear();
62 compressed_data_.outer_ptr.assign(outer_size + 1, 0); // initialise outer_ptr with zeroes
63
64 // 1. Counts the values in each row/column
65 for (const auto& [key, val] : sparse_data_) {
66 size_t outer = isRowMajor ? key[0] : key[1]; // if CSR uses the row as outer if CSC uses the column
67 compressed_data_.outer_ptr[outer + 1]++; // to know how many elements are in each column/row
68 }
69
70 // 2. Indicates where starts each row
71 for (size_t i = 1; i <= outer_size; ++i) { // outer_ptr[i] -> index of starting of the i-th row/column in the vector values
72 // outer_ptr[i+1] -> index excluded of the next row
73 compressed_data_.outer_ptr[i] += compressed_data_.outer_ptr[i - 1];
74 }
75
76 // 3. Allocates space and to not call pushback each time
77 size_t nnz = sparse_data_.size(); // total number of non zero values
78 compressed_data_.values.resize(nnz);
79 compressed_data_.inner_index.resize(nnz);
80 std::vector<size_t> temp_offset = compressed_data_.outer_ptr;
81
82 // 4. Giving values to the vectors
83 for (const auto& [key, val] : sparse_data_) {
84 size_t outer = isRowMajor ? key[0] : key[1];
85 size_t inner = isRowMajor ? key[1] : key[0];
86 size_t idx = temp_offset[outer]++;
87
88 compressed_data_.values[idx] = val;
89 compressed_data_.inner_index[idx] = inner;
90 }
91
92 sparse_data_.clear();
93
94}

96template<typename T, StorageOrder Order>
97void Matrix<T, Order>::decompress() {
98// Converts the matrix from compressed form (CSR/CSC) to uncompressed form (Coo-MAP).
99// It reconstructs sparse_data_ using the compressed_data_ arrays and clears the compressed storage afterward.
100
101 sparse_data_.clear();
102
103 constexpr bool isRowMajor = (Order == StorageOrder::RowMajor);
104 size_t outer_size = compressed_data_.outer_ptr.size() - 1;
105
106 for (size_t outer = 0; outer < outer_size; ++outer) {
107 for (size_t k = compressed_data_.outer_ptr[outer]; k < compressed_data_.outer_ptr[outer + 1]; ++k) {
108 size_t inner = compressed_data_.inner_index[k];
109
110 size_t i = isRowMajor ? outer : inner;
111 size_t j = isRowMajor ? inner : outer;
112
113 update(i, j, compressed_data_.values[k]); // utilizzo del metodo update()
114 }
115 }
116
117 compressed_data_.clear();
118}

120template<typename T, StorageOrder Order>
121template<NormType norm_type>
122T Matrix<T, Order>::norm() {
123// Computes the matrix norm (One, Infinity, or Frobenius) depending on the template parameter.
124// Handles both compressed (CSR/CSC) and uncompressed (COOmap) storage formats, with automatic transposition if needed.
125// Returns a scalar value representing the computed norm.
126
127 T norm = T(0);
128 if (is_compressed()){
129 // Compressed case - CSR/CSC
130
131 bool was_transposed = false;
132 if constexpr (norm_type == NormType::One) {
133 if constexpr (Order == StorageOrder::RowMajor){
134 transpose();
135 was_transposed=true;
136 }
137 for (size_t j = 0; j < cols_; ++j) {
138 T col_sum = T(0);
139 for (size_t idx = compressed_data_.outer_ptr[j]; idx < compressed_data_.outer_ptr[j+1]; ++idx) {
140 col_sum += std::abs(compressed_data_.values[idx]);
141 }
142 if (std::abs(col_sum) > std::abs(norm)) {norm = col_sum;}
143 }
144 if(was_transposed){transpose();}
145 return norm;
146
147 } else if constexpr (norm_type == NormType::Infinity) {
148 if constexpr (Order == StorageOrder::ColumnMajor){
149 transpose();
150 was_transposed=true;
151 }
152 for (size_t j = 0; j < cols_; ++j) {
153 T row_sum = T(0);
154 for (size_t idx = compressed_data_.outer_ptr[j]; idx < compressed_data_.outer_ptr[j+1]; ++idx) {
155 row_sum += std::abs(compressed_data_.values[idx]);
156 }
157 if (std::abs(row_sum) > std::abs(norm)) {norm = row_sum;}
158 }
159 if(was_transposed){transpose();}
160 return norm;
161 } else if constexpr (norm_type == NormType::Frobenius) {
162 if constexpr (Order == StorageOrder::ColumnMajor){
163 transpose();
164 was_transposed=true;
165 }
166 for (size_t j = 0; j < cols_; ++j) {
167 T row_sum = T(0);
168 for (size_t idx = compressed_data_.outer_ptr[j]; idx < compressed_data_.outer_ptr[j+1]; ++idx) {
169 row_sum += std::pow(std::abs(compressed_data_.values[idx]),T(2));
170 }
171 norm += row_sum;
172 }
173 if(was_transposed){transpose();}
174 return std::sqrt(norm);
175 }
176 }
177 else {
178 // TO DO
179 // Uncompressed case - COOmap
180 if constexpr (norm_type == NormType::One) {
181 std::vector<T> col_sums(cols_, T(0)); // empty vector of dimension cols_
182 for(const auto& [coord, value] : sparse_data_) {
183 col_sums[coord[1]] += std::abs(value);
184 }
185 for (const auto& sum : col_sums) {
186 if (std::abs(sum) > std::abs(norm)) { norm = sum; }
187 // needed to support also std::complex...
188 }
189 return norm;
190 } else if constexpr (norm_type == NormType::Infinity) {
191 std::vector<T> row_sums(rows_, T(0)); // empty vector of dimension rows_
192 for(const auto& [coord, value] : sparse_data_) {
193 row_sums[coord[0]] += std::abs(value);
194 }
195 for (const auto& sum : row_sums) {
196 if (std::abs(sum) > std::abs(norm)) { norm = sum; }
197 // needed to support also std::complex...
198 }
199 return norm;
200 } else if constexpr (norm_type == NormType::Frobenius) {
201 for(const auto& [coord, value] : sparse_data_){
202 norm += std::pow(std::abs(value), T(2));
203 }
204 return std::sqrt(norm);
205 }
206 }

210template<typename T, StorageOrder Order>
211void Matrix<T, Order>::transpose() {
212// Transposes the matrix in-place by swapping rows and columns.
213// Updates the sparse data (non-zero entries) by flipping their indices.
214// Recompresses the matrix if it was previously compressed before transposing.
215
216 // If compressed, decompress first
217 bool was_compressed = false;
218 if (is_compressed()) {
219 decompress();
220 was_compressed = true;
221 }
222
223 // New sparse_data where we will store transposed entries
224 std::map<std::array<size_t, 2>, T> new_sparse_data;
225
226 for (const auto& [key, value] : sparse_data_) {
227 std::array<size_t, 2> transposed_key = {key[1], key[0]}; // flip (i,j) -> (j,i)
228 new_sparse_data[transposed_key] = value;
229 }
230
231 // Swap rows and columns
232 std::swap(rows_, cols_);
233
234 // Replace old sparse data with the transposed one
235 sparse_data_ = std::move(new_sparse_data);
236
237 // If it was compressed in the beginning, recompress it
238 if(was_compressed){compress();}
239}

242template<typename T, StorageOrder Order>
243std::vector<T> Matrix<T, Order>::compressed_product_by_vector_parallel(const std::vector<T>& v) const {
244// Multiplies a compressed matrix by a vector v using parallelization for faster computation.
245// Uses OpenMP to parallelize the dot product calculation for each row of the matrix.
246// Inputs: v - a vector of type T, Outputs: a vector of type T with the result of the multiplication.
247
248 std::vector<T> output(rows_, T(0));
249
250 if constexpr (Order == StorageOrder::RowMajor) {
251 // RowMajor (CSR): traverse row by row (parallelized)
252 #pragma omp parallel for
253 for (size_t i = 0; i < rows_; ++i) {
254 T sum = 0;
255 for (size_t k = compressed_data_.outer_ptr[i]; k < compressed_data_.outer_ptr[i + 1]; ++k) {
256 sum += compressed_data_.values[k] * v[compressed_data_.inner_index[k]];
257 }
258 output[i] = sum;
259 }
260 } else {
261 // ColumnMajor (CSC): traverse column by column
262 #pragma omp parallel for
263 for (size_t j = 0; j < cols_; ++j) {
264 for (size_t k = compressed_data_.outer_ptr[j]; k < compressed_data_.outer_ptr[j + 1]; ++k) {
265 size_t i = compressed_data_.inner_index[k];
266 #pragma omp atomic
267 output[i] += compressed_data_.values[k] * v[j];
268 }
269 }
270 }
271
272 return output;
273}

276std::vector<T> Matrix<T, Order>::compressed_product_by_vector(const std::vector<T>& v) const {
277 std::vector<T> output(rows_, T(0));
278
279 if constexpr (Order == StorageOrder::RowMajor) {
280 // RowMajor (CSR): traverse row by row
281 for (size_t i = 0; i < rows_; ++i) {
282 T sum = 0;
283 for (size_t k = compressed_data_.outer_ptr[i]; k < compressed_data_.outer_ptr[i + 1]; ++k) {

301template<typename T, StorageOrder Order>
302std::vector<T> Matrix<T, Order>::product_by_vector(const std::vector<T>& v) const {
303// Multiplies the matrix by a vector v. If the matrix is compressed, it uses either parallel or regular multiplication based on the number of rows (CSR case) or columns (CSC case).
304// For uncompressed matrices in COO format, it performs the multiplication by iterating over sparse data.
305// Inputs: v - a vector of type T, Outputs: a vector of type T with the result of the multiplication.
306
307 if (is_compressed()) {
308 if constexpr (Order==StorageOrder::RowMajor){
309 if (rows_>=params::NROWS_PARALLELIZATON_LIMIT){
310 return compressed_product_by_vector_parallel(v);
311 }
312 else{
313 return compressed_product_by_vector(v);
314 }
315 }
316 else{
317 if (rows_>=params::NCOLS_PARALLELIZATON_LIMIT){
318 return compressed_product_by_vector_parallel(v);
319 }
320 else{
321 return compressed_product_by_vector(v);
322 }
323 }
324
325 }
326 else {
327 // Create the output vector
328 std::vector<T> output(rows_, 0);
329
330 // Uncompressed multiplication (COO)
331 for (const auto& [key, val] : sparse_data_) {
332 size_t i = key[0];
333 size_t j = key[1];
334 output[i] += val * v[j];
335 }
336 return output;
337 }

341template<typename T, StorageOrder Order> // This function applies to any Matrix type with any storage order
342std::vector<T> Matrix<T, Order>::operator*(const Matrix<T, Order>& rhs) const {// The result is a plain std::vector<T>, representing the product result
343// Overloads the * operator to perform matrix-vector multiplication when rhs is a column vector matrix.
344// Assumes rhs is effectively a column vector; converts it to std::vector<T> and uses product_by_vector for the computation.
345// Ensures rhs is in uncompressed form before extracting values.
346
347 //Check dimension compatibility
348 if (cols_ != rhs.rows_) {
349 throw std::invalid_argument("Matrix dimensions do not match for multiplication.");
350 }
351
352 //Convert rhs into a std::vector<T>
353 std::vector<T> vec(rhs.rows_, 0);
354
355 // Make sure rhs is decompressed
356 if (rhs.is_compressed()) {
357 Matrix<T, Order> rhs_copy = rhs;// We cannot modify rhs (it's a const reference), so we create a copy
358 rhs_copy.decompress();
359 for (const auto& [key, val] : rhs_copy.sparse_data_) {
360 vec[key[0]] = val; // key[0] represents the row index (since rhs is a column vector)
361 }
362 } else {
363 for (const auto& [key, val] : rhs.sparse_data_) {
364 vec[key[0]] = val; // If already uncompressed, directly extract values from sparse_data_
365 }
366 }
367 // Use existing method to perform matrix-vector multiplication
368 return this->product_by_vector(vec);
369}

372template<typename T, StorageOrder Order>
373bool Matrix<T, Order>::mm_stringstream_to_sparsedata_loader(const std::istringstream& iss_original){
374// Parses a Matrix Market file's content (from a stringstream) into sparse matrix data format.
375// Reads the header to get matrix dimensions, then processes triplets (row, column, value) to populate the matrix.
376// Inputs: iss_original - an input stringstream containing the file content, Outputs: true if parsing is successful, false otherwise.
377
378 std::istringstream iss(iss_original.str()); // fai una copia per poter usare getline e >> separatamente
379 std::string line;
380 int max_row_idx = 0;
381 int max_col_idx = 0;
382
383 // Skip header and comments
384 while (std::getline(iss, line)) {
385 if (line.empty() || line[0] == '%') continue;
386 else break; // first non-comment line
387 }
388
389 // parse dimensions
390 std::istringstream header_line(line);
391 size_t rows, cols, entries;
392 if (!(header_line >> rows >> cols >> entries)) {
393 std::cerr << "Error parsing header line: " << line << std::endl;
394 return false;
395 }
396
397 // parse triplets
398 int row, col;
399 double value;
400 while (iss >> row >> col >> value) {
401 if (row > max_row_idx) max_row_idx = row;
402 if (col > max_col_idx) max_col_idx = col;
403 update(row - 1, col - 1, value); // Matrix Market is 1-based
404 }
405
406 rows_ = static_cast<size_t>(max_row_idx);
407 cols_ = static_cast<size_t>(max_col_idx);
408
409 return true;
410}

412template<typename T, StorageOrder Order>
413std::string Matrix<T, Order>::mm_extract_gz(const std::string& filename){
414// Extracts and reads the contents of a compressed (.gz) Matrix Market file.
415// Uses zlib's gz functions to read the file in chunks and accumulate its contents into a string.
416// Inputs: filename - the file path, Outputs: the content of the file as a string.
417
418 gzFile file = gzopen(filename.c_str(), "rb");
419 if (!file) {
420 std::cerr << "Error: could not open file " << filename << std::endl;
421 return "";
422 }
423
424
425
426 char buffer[params::BUFFER_SIZE];
427
428 std::string file_content;
429
430 int bytes_read;
431 while ((bytes_read = gzread(file, buffer, params::BUFFER_SIZE - 1)) > 0) {
432 buffer[bytes_read] = '\0'; // Null-terminate buffer
433 file_content += buffer; // Accumulate contents
434 }
435
436 gzclose(file);
437 return file_content;

440template<typename T, StorageOrder Order>
441bool Matrix<T, Order>::mm_load_mtx(const std::string& filename){
442// Loads a Matrix Market (.mtx or .mtx.gz) file and parses its contents into sparse data format.
443// Supports both compressed (.mtx.gz) and uncompressed (.mtx) Matrix Market files.
444// Inputs: filename - the file path, Outputs: true if loading is successful, false otherwise.
445
446
447 if (filename.ends_with(".mtx.gz")) {
448 sparse_data_.clear(); // clear sparse_data_ values
449 compressed_data_.clear(); // clear compressed data values
450
451 auto file_content = mm_extract_gz(filename);

463 }
464
465 // Leggi il contenuto del file direttamente in un istringstream

467 iss.str(std::string((std::istreambuf_iterator<char>(ifs)), std::istreambuf_iterator<char>()));
468
469 return mm_stringstream_to_sparsedata_loader(iss);

478template<typename T, StorageOrder Order>
479void Matrix<T, Order>::resize(size_t new_rows, size_t new_cols) {
480// Resizes the matrix to new_rows x new_cols, updating internal dimensions.
481// Removes all elements outside the new bounds.
482// INPUT: new_rows (size_t), new_cols (size_t) - new dimensions; OUTPUT: none (matrix modified in-place).
483
484 bool was_compressed = false;
485 if (is_compressed()){
486 decompress();
487 was_compressed = true;
488 }
489 rows_ = new_rows;
490 cols_ = new_cols;
491
492 // Remove entries that are now out of bounds
493 for (auto it = sparse_data_.begin(); it != sparse_data_.end(); ) {
494 auto [row, col] = it->first;
495 if (row >= new_rows || col >= new_cols) {
496 it = sparse_data_.erase(it); // erase returns the next valid iterator
497 } else {
498 ++it;
499 }
500 }
501
502 if (was_compressed){
503 compress();
504 }
505}

507template<typename T, StorageOrder Order>
508std::vector<T> Matrix<T, Order>::diagonal_view() const {
509// Returns a vector containing the diagonal elements of the matrix.
510// If an element on the diagonal is not stored explicitly (i.e., zero in sparse form), we assume it is 0.
511
512 std::vector<T> diag(std::min(rows_, cols_), T(0));
513
514 if (!is_compressed()) {
515 // Uncompressed (COO map) case
516 for (size_t i = 0; i < diag.size(); ++i) {
517 std::array<size_t, 2> key = {i, i};
518 auto it = sparse_data_.find(key);
519 if (it != sparse_data_.end()) {
520 diag[i] = it->second;
521 }
522 }
523 } else {
524 // Compressed (CSR/CSC) case
525 constexpr bool isRowMajor = (Order == StorageOrder::RowMajor);
526
527 if (isRowMajor) {
528 for (size_t i = 0; i < diag.size(); ++i) {
529 for (size_t k = compressed_data_.outer_ptr[i]; k < compressed_data_.outer_ptr[i + 1]; ++k) {
530 if (compressed_data_.inner_index[k] == i) {
531 diag[i] = compressed_data_.values[k];
532 break;
533 }
534 }
535 }
536 } else {
537 for (size_t j = 0; j < diag.size(); ++j) {
538 for (size_t k = compressed_data_.outer_ptr[j]; k < compressed_data_.outer_ptr[j + 1]; ++k) {
539 if (compressed_data_.inner_index[k] == j) {
540 diag[j] = compressed_data_.values[k];
541 break;
542 }
543 }
544 }
545 }
546 }
547
548 return diag;
549}

554template<typename T, StorageOrder Order>
555void Matrix<T, Order>::print(int width) const {
556 // Prints the matrix in a tabular, human-readable form.
557 // If the matrix is uncompressed, it prints from sparse_data_.
558 // If compressed, it reconstructs the row-wise representation using compressed_data_.
559 // Parameters:
560 // width (optional): number of characters per column (default = 6, see Matrix.hpp).
561
562 if (!is_compressed()) {
563 // Uncompressed mode
564 for (std::size_t i = 0; i < rows_; ++i) {
565 for (std::size_t j = 0; j < cols_; ++j) {
566 std::array<size_t, 2> key = {i, j};
567 if (sparse_data_.count(key)) {
568 std::cout << std::setw(width) << sparse_data_.at(key);
569 } else {
570 std::cout << std::setw(width) << T(0);
571 }
572 }
573 std::cout << '\n';
574 }
575 } else {
576 // Compressed mode
577 for (size_t i = 0; i < compressed_data_.outer_ptr.size() - 1; ++i) {
578 auto row_start = compressed_data_.outer_ptr[i];
579 auto row_end = compressed_data_.outer_ptr[i + 1];
580
581 std::unordered_map<size_t, T> row_values;
582 for (size_t idx = row_start; idx < row_end; ++idx) {
583 row_values[compressed_data_.inner_index[idx]] = compressed_data_.values[idx];
584 }
585
586 for (size_t j = 0; j < cols_; ++j) {
587 if (row_values.count(j)) {
588 std::cout << std::setw(width) << row_values[j];
589 } else {
590 std::cout << std::setw(width) << T(0);
591 }
592 }
593 std::cout << '\n';
594 }
595 }

645template<typename T, StorageOrder Order>
646size_t Matrix<T, Order>::weight() const{
647// Calculates the memory usage (weight) of the matrix based on its storage format.
648// For compressed matrices, it sums the sizes of the values, indices, and pointers; for uncompressed, it estimates based on the sparse data structure.
649// Outputs: the memory size in bytes.
650
651 if (is_compressed()){
652 return compressed_data_.values.size() * sizeof(T)+ compressed_data_.inner_index.size() * sizeof(size_t)+ compressed_data_.outer_ptr.size() * sizeof(size_t);
653
654 }
655 else{
656 size_t size_per_element = sizeof(std::array<size_t, 2>) + sizeof(T) + 3 * sizeof(void*) + sizeof(bool);
657 return sizeof(sparse_data_) + sparse_data_.size() * size_per_element;
658 }
659}

663// Checks whether the matrix is in compressed form (CSR/CSC).
664// Returns true if all three components of compressed_data_ are non-empty.
665
